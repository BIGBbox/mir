[@获取角色属性]
{

#if
#act
CreateFile ..\QuestDiary\排行榜\排行榜数据.txt
#if

#act
SortHumVar 角色攻击力 S$角色名字 N$角色攻击力 0 1 50

[@写入排行榜]

#if
#act

WriteConfigFileItem ..\QuestDiary\排行榜\排行榜数据.txt <$USERNAME> 角色等级 <$LEVEL>

[@编辑排行榜文件]
#if
#act
GetRandomText ..\QuestDiary\排行榜\排行榜数据.txt S10 0
;SendMsg 5 取出的字符串是:<$STR(S10)>::0

#if
not equal <$str(S10)>
#act
inc N$行号 1

goto @编辑排行榜文件
; #elseact
; goto @写入排行榜
